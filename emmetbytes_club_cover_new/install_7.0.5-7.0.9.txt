2. Paste this module inside the EmmetBytes directory ( {your_websites_root_path}/modules/EmmetBytes/emmetbytes_club_cover )
3. Install the module

            // BOF THE EMMETBYTES CLUB COVER METHODS
            // setting emmetbytes club cover avatar
            private function setEmmetBytesAvatar($sImg){
                $this->clubCoverTmpPath = BX_AVA_DIR_TMP . $this->_iProfileId . 'eb' . BX_AVA_EXT;
                $o = BxDolImageResize::instance(EB_AVA_W, EB_AVA_W);
                $o->removeCropOptions ();
                $o->setJpegOutput (true);
                $o->setSize (EB_AVA_W, EB_AVA_W);
                $o->setSquareResize (true);
                $o->setJpegQuality (getParam('bx_avatar_quality'));
                if (IMAGE_ERROR_SUCCESS != $o->resize($sImg, $this->clubCoverTmpPath))
                    return false;
            }

            // rename emmetbytes club cover avatar
            private function renameEmmetBytesClubCoverAvatar($iAvatar){
                if(isset($this->clubCoverTmpPath) && !empty($this->clubCoverTmpPath)){
                    $sEmmetBytesImagePath = BX_AVA_DIR_USER_AVATARS . $iAvatar . 'eb' . BX_AVA_EXT;
                    copy($this->clubCoverTmpPath, $sEmmetBytesImagePath);
                }
            }
            // EOF THE EMMETBYTES CLUB COVER METHODS
